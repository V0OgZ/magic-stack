<script>
const adapter = new V2Adapter();

// Tick obligatoire
setInterval(() => {
  adapter.tick({
    temporal: { tw: Date.now(), te: Date.now() },
    entities: placedIcons
  });
}, 100);

// Sauvegarder avec API V2
function saveMap() {
  const mapData = {
    position: {x:0, y:0, z:0, t:0, c:1, psi:0.5},
    icons: placedIcons
  };
  
  fetch('http://localhost:8080/api/interstice/upload', {
    method: 'POST',
    body: JSON.stringify(mapData)
  });
}
</script>
